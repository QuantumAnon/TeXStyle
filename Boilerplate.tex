    %Commands
  
      %Column Vectors
        \newcount\colveccount
        \newcommand*\colvec[1]{
          \global\colveccount#1
          \begin{bmatrix}
          \colvecnext
        }
        \def\colvecnext#1{
          #1
          \global\advance\colveccount-1
          \ifnum\colveccount>0
            \\
            \expandafter\colvecnext
          \else
            \end{bmatrix}
          \fi
        }
  
      %Row Vectors
        \newcount\rowveccount
        \newcommand*\rowvec[1]{
          \global\rowveccount#1
          \begin{bmatrix}
          \rowvecnext
        }
        \def\rowvecnext#1{
          #1
          \global\advance\rowveccount-1
          \ifnum\rowveccount>0
            &
            \expandafter\colvecnext
          \else
            \\
            \end{bmatrix}
          \fi
        }
  
      %Norm vectors
        \newcount\normveccount
        \newcommand*\normvec[1]{
          \global\normveccount#1
          \begin{Vmatrix}
          \normvecnext
        }
  
        \def\normvecnext#1{
          #1
          \global\advance\normveccount-1
          \ifnum\normveccount>0
            \\
            \expandafter\normvecnext
          \else
            \end{Vmatrix}
          \fi
        }
